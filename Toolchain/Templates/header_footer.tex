\usepackage{fancyhdr}
\usepackage{graphicx}

\setlength{\headheight}{20pt}

\pagestyle{fancy}
\fancyhf{}

% Professional header and footer styling
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0.5pt}
\renewcommand{\headrule}{\hbox to\headwidth{\color{bordergray}\leaders\hrule height \headrulewidth\hfill}}
\renewcommand{\footrule}{\hbox to\headwidth{\color{bordergray}\leaders\hrule height \footrulewidth\hfill}}

% Header: logo a sinistra, titolo a destra
\fancyhead[L]{%
  \ifdefined\BLDLogo
    \ifx\BLDLogo\empty
      % niente logo
    \else
      \includegraphics[height=18pt]{\BLDLogo}%
    \fi
  \fi
}
\fancyhead[R]{%
  \ifdefined\BLDTitle
    {\footnotesize\color{textgray}\BLDTitle}%
  \fi
}

% Footer con design elegante
\fancyfoot[L]{%
  \ifdefined\BLDVersion
    {\footnotesize\color{textgray}Version \BLDVersion}%
  \fi
}
\fancyfoot[C]{%
  {\footnotesize\color{textgray}\thepage}%
}
\fancyfoot[R]{%
  \ifdefined\BLDUpdated
    {\footnotesize\color{textgray}\BLDUpdated}%
  \fi
}

% Stile "plain" per TOC e cover pages
\fancypagestyle{plain}{%
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0.5pt}
  \renewcommand{\footrule}{\hbox to\headwidth{\color{bordergray}\leaders\hrule height \footrulewidth\hfill}}
  \fancyfoot[L]{%
    \ifdefined\BLDVersion
      {\footnotesize\color{textgray}Version \BLDVersion}%
    \fi
  }
  \fancyfoot[C]{%
    {\footnotesize\color{textgray}\thepage}%
  }
  \fancyfoot[R]{%
    \ifdefined\BLDUpdated
      {\footnotesize\color{textgray}\BLDUpdated}%
    \fi
  }
}
